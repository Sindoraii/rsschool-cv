function BubbleSort(arr) {
  let localArr = [...arr];
  let flag = true;
  while (flag) {
    flag = false;
    for (let i = 0; i < localArr.length; i++) {
      if (localArr[i] > localArr[i + 1]) {
        let saveValue = localArr[i];
        localArr[i] = localArr[i + 1];
        localArr[i + 1] = saveValue;
        flag = true;
      }
    }
  }
  return localArr;
}
